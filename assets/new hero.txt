        class Particle {
            constructor() {
                this.x = Math.random() * canvas.width;
                this.y = Math.random() * canvas.height;
                this.baseX = this.x;
                this.baseY = this.y;
                this.vx = (Math.random() - 0.5) * 0.3;
                this.vy = (Math.random() - 0.5) * 0.3;
                this.radius = Math.random() * 2 + 1;
                this.density = Math.random() * 30 + 1;
            }

            update() {
                // Mouse interaction
                if (mouse.x != null && mouse.y != null) {
                    let dx = mouse.x - this.x;
                    let dy = mouse.y - this.y;
                    let dist = Math.sqrt(dx * dx + dy * dy);
                    let forceX = dx / dist;
                    let forceY = dy / dist;
                    let maxDist = mouse.radius;
                    let force = (maxDist - dist) / maxDist;

                    if (dist < mouse.radius) {
                        this.x -= forceX * force * this.density * 0.5;
                        this.y -= forceY * force * this.density * 0.5;
                    }
                }

                // Normal movement
                this.x += this.vx;
                this.y += this.vy;

                // Slowly return to base
                this.x += (this.baseX - this.x) * 0.01;
                this.y += (this.baseY - this.y) * 0.01;

                // Bounce off edges
                if (this.x < 0 || this.x > canvas.width) {
                    this.vx *= -1;
                    this.baseX = this.x;
                }
                if (this.y < 0 || this.y > canvas.height) {
                    this.vy *= -1;
                    this.baseY = this.y;
                }
            }

            draw() {
                // Lighter colors - cyan/turquoise tones
                let brightness = 0.6;
                if (mouse.x != null && mouse.y != null) {
                    let dx = mouse.x - this.x;
                    let dy = mouse.y - this.y;
                    let dist = Math.sqrt(dx * dx + dy * dy);
                    if (dist < mouse.radius) {
                        brightness = 1 - (dist / mouse.radius) * 0.4;
                    }
                }
                
                ctx.beginPath();
                ctx.arc(this.x, this.y, this.radius, 0, Math.PI * 2);
                ctx.fillStyle = `rgba(100, 200, 255, ${0.4 * brightness})`;
                ctx.fill();
            }
        }

        const particles = [];
        const count = 150;
        
        function initParticles() {
            particles.length = 0;
            for (let i = 0; i < count; i++) {
                particles.push(new Particle());
            }
        }
        
        initParticles();

        function drawConnections() {
            for (let i = 0; i < particles.length; i++) {
                for (let j = i + 1; j < particles.length; j++) {
                    let dx = particles[i].x - particles[j].x;
                    let dy = particles[i].y - particles[j].y;
                    let dist = Math.sqrt(dx * dx + dy * dy);

                    if (dist < 140) {
                        let opacity = 0.15 * (1 - dist / 140);
                        
                        // Brighter near mouse
                        if (mouse.x != null && mouse.y != null) {
                            let midX = (particles[i].x + particles[j].x) / 2;
                            let midY = (particles[i].y + particles[j].y) / 2;
                            let mouseDist = Math.sqrt((mouse.x - midX) ** 2 + (mouse.y - midY) ** 2);
                            if (mouseDist < mouse.radius) {
                                opacity *= 1 + (1 - mouseDist / mouse.radius) * 2;
                            }
                        }
                        
                        ctx.beginPath();
                        ctx.strokeStyle = `rgba(100, 200, 255, ${opacity})`;
                        ctx.lineWidth = 0.6;
                        ctx.moveTo(particles[i].x, particles[i].y);
                        ctx.lineTo(particles[j].x, particles[j].y);
                        ctx.stroke();
                    }
                }
            }
        }

            if (mouse.x != null && mouse.y != null) {

                gradient.addColorStop(0.5, 'rgba(139, 0, 255, 0.05)');
                gradient.addColorStop(1, 'rgba(0, 0, 0, 0)');
                ctx.beginPath();
                ctx.arc(mouse.x, mouse.y, mouse.radius, 0, Math.PI * 2);
                ctx.fillStyle = gradient;
                ctx.fill();
            }
        }

        function visualizeAudio() {
            const bufferLength = analyser.frequencyBinCount;
            const dataArray = new Uint8Array(bufferLength);
            const centerX = 150;
            const centerY = 150;
            
            function draw() {
                if (!isListening) return;
                animationId = requestAnimationFrame(draw);
                
                analyser.getByteFrequencyData(dataArray);
                audioCtx2.clearRect(0, 0, 300, 300);
                
                const avgVolume = dataArray.reduce((a, b) => a + b) / bufferLength;
                
                const nodeCount = 8;
                const nodes = [];
                for (let i = 0; i < nodeCount; i++) {
                    const angle = (i / nodeCount) * Math.PI * 2 - Math.PI / 2;
                    const radius = 90 + (dataArray[i * 12] / 255) * 25;
                    nodes.push({
                        x: centerX + Math.cos(angle) * radius,
                        y: centerY + Math.sin(angle) * radius,
                        size: 3 + (dataArray[i * 12] / 255) * 4
                    });
                }
                
                // Draw connections
                for (let i = 0; i < nodes.length; i++) {
                    for (let j = i + 1; j < nodes.length; j++) {
                        const opacity = 0.08 + (dataArray[(i + j) * 5] / 255) * 0.2;
                        audioCtx2.beginPath();
                        audioCtx2.strokeStyle = `rgba(139, 0, 255, ${opacity})`;
                        audioCtx2.lineWidth = 1;
                        audioCtx2.moveTo(nodes[i].x, nodes[i].y);
                        audioCtx2.lineTo(nodes[j].x, nodes[j].y);
                        audioCtx2.stroke();
                    }
                    
                    const centerOpacity = 0.15 + (dataArray[i * 8] / 255) * 0.25;
                    audioCtx2.beginPath();
                    audioCtx2.strokeStyle = `rgba(0, 153, 255, ${centerOpacity})`;
                    audioCtx2.lineWidth = 1.5;
                    audioCtx2.moveTo(nodes[i].x, nodes[i].y);
                    audioCtx2.lineTo(centerX, centerY);
                    audioCtx2.stroke();
                }
                
                // Draw nodes
                nodes.forEach((node, i) => {
                    const gradient = audioCtx2.createRadialGradient(node.x, node.y, 0, node.x, node.y, node.size * 2);
                    gradient.addColorStop(0, `rgba(139, 0, 255, ${0.6 + (dataArray[i * 12] / 255) * 0.2})`);
                    gradient.addColorStop(1, 'rgba(139, 0, 255, 0)');
                    
                    audioCtx2.beginPath();
                    audioCtx2.arc(node.x, node.y, node.size, 0, Math.PI * 2);
                    audioCtx2.fillStyle = `rgba(100, 200, 255, ${0.5 + (dataArray[i * 12] / 255) * 0.3})`;
                    audioCtx2.fill();
                    
                    audioCtx2.beginPath();
                    audioCtx2.arc(node.x, node.y, node.size * 2, 0, Math.PI * 2);
                    audioCtx2.fillStyle = gradient;
                    audioCtx2.fill();
                });
                
                // Outer ring
                audioCtx2.beginPath();
                audioCtx2.arc(centerX, centerY, 75 + avgVolume * 0.15, 0, Math.PI * 2);
                audioCtx2.strokeStyle = `rgba(139, 0, 255, ${0.15 + (avgVolume / 255) * 0.2})`;
                audioCtx2.lineWidth = 2;
                audioCtx2.stroke();
            }
            
            draw();
        }

        function visualizeAudioFallback() {
            let t = 0;
            function draw() {
                if (!audioOverlay.classList.contains('active')) return;
                requestAnimationFrame(draw);
                t += 0.008;
                
                audioCtx2.clearRect(0, 0, 300, 300);
                const centerX = 150, centerY = 150;
                
                for (let i = 0; i < 8; i++) {
                    const angle = (i / 8) * Math.PI * 2 + t;
                    const radius = 90 + Math.sin(t * 1.5 + i) * 15;
                    const x = centerX + Math.cos(angle) * radius;
                    const y = centerY + Math.sin(angle) * radius;
                    
                    audioCtx2.beginPath();
                    audioCtx2.arc(x, y, 4, 0, Math.PI * 2);
                    audioCtx2.fillStyle = `rgba(100, 200, 255, ${0.4 + Math.sin(t + i) * 0.2})`;
                    audioCtx2.fill();
                    
                    audioCtx2.beginPath();
                    audioCtx2.strokeStyle = `rgba(139, 0, 255, 0.2)`;
                    audioCtx2.moveTo(x, y);
                    audioCtx2.lineTo(centerX, centerY);
                    audioCtx2.stroke();
                }
            }
            draw();
        }
